Multi Select / Single Select

Note: https://www.npmjs.com/package/ng-multiselect-dropdown

Instructions:

  • Run command

    npm install ng-multiselect-dropdown --save

  • import modules in app.module.ts

    import {NgMultiSelectDropDownModule} from "ng-multiselect-dropdown";

    add below provider in app.module.ts in providers array

    NgMultiSelectDropDownModule.forRoot()

    Component should look like this

    export class AppComponent implements OnInit {
    dropdownList = [];
    selectedItems = [];
    dropdownSettings = {};
    ngOnInit() {
        this.dropdownList = [
            { item_id: 1, item_text: 'Mumbai' },
            { item_id: 2, item_text: 'Bangaluru' },
            { item_id: 3, item_text: 'Pune' },
            { item_id: 4, item_text: 'Navsari' },
            { item_id: 5, item_text: 'New Delhi' }
        ];
        this.selectedItems = [
            { item_id: 3, item_text: 'Pune' },
            { item_id: 4, item_text: 'Navsari' }
        ];
        this.dropdownSettings:IDropdownSettings = {
            singleSelection: false,
            idField: 'item_id',
            textField: 'item_text',
            selectAllText: 'Select All',
            unSelectAllText: 'UnSelect All',
            itemsShowLimit: 3,
            allowSearchFilter: true
        };
    }
    onItemSelect(item: any) {
    console.log(item);
    }
    onSelectAll(items: any) {
    console.log(items);
    }
}

  • --> copy/paste below html in your component file

    <ng-multiselect-dropdown
    [placeholder]="'custom placeholder'"
    [data]="dropdownList"
    [(ngModel)]="selectedItems"
    [settings]="dropdownSettings"
    (onSelect)="onItemSelect($event)"
    (onSelectAll)="onSelectAll($event)">
</ng-multiselect-dropdown>
